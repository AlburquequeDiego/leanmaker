-- 1. VERIFICAR TODOS LOS USUARIOS EN LA TABLA PRINCIPAL
SELECT 
    id,
    email,
    first_name,
    last_name,
    role,
    is_verified,
    is_active,
    date_joined
FROM users_usuario 
WHERE email LIKE '%@uchile.cl' 
   OR email LIKE '%@uc.cl'
   OR email LIKE '%@udec.cl'
   OR email LIKE '%@usm.cl'
   OR email LIKE '%@usach.cl'
   OR email LIKE '%@uach.cl'
   OR email LIKE '%@uv.cl'
   OR email LIKE '%@ufro.cl'
   OR email LIKE '%@utalca.cl'
   OR email LIKE '%@uantof.cl'
   OR email = 'admin@leanmaker.cl'
ORDER BY email;

-- 3. VERIFICAR ESTUDIANTES ESPECÍFICOS
SELECT 
    u.id,
    u.email,
    u.first_name,
    u.last_name,
    u.role,
    e.career,
    e.semester,
    e.strikes
FROM users_usuario u
LEFT JOIN students_estudiante e ON u.id = e.user_id
WHERE u.role = 'student'
  AND (u.email LIKE '%@uchile.cl' 
   OR u.email LIKE '%@uc.cl'
   OR u.email LIKE '%@udec.cl'
   OR u.email LIKE '%@usm.cl'
   OR u.email LIKE '%@usach.cl'
   OR u.email LIKE '%@uach.cl'
   OR u.email LIKE '%@uv.cl'
   OR u.email LIKE '%@ufro.cl'
   OR u.email LIKE '%@utalca.cl'
   OR u.email LIKE '%@uantof.cl')
ORDER BY u.email;

-- 4. VERIFICAR EMPRESAS ESPECÍFICAS
SELECT 
    u.id,
    u.email,
    u.first_name,
    u.last_name,
    u.role,
    e.company_name,
    e.industry,
    e.size,
    e.website
FROM users_usuario u
LEFT JOIN companies_empresa e ON u.id = e.user_id
WHERE u.role = 'company'
  AND (u.email LIKE '%@uchile.cl' 
   OR u.email LIKE '%@uc.cl'
   OR u.email LIKE '%@udec.cl'
   OR u.email LIKE '%@usm.cl'
   OR u.email LIKE '%@usach.cl'
   OR u.email LIKE '%@uach.cl'
   OR u.email LIKE '%@uv.cl'
   OR u.email LIKE '%@ufro.cl'
   OR u.email LIKE '%@utalca.cl'
   OR u.email LIKE '%@uantof.cl')
ORDER BY u.email;
